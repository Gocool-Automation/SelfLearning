Fibonacci Using while loop

int num1 = 0, num2 = 1, n = 5;
int num3;
int counter = 0;
while(counter < n) {
System.out.print(num1 + " ");
num3 = num2 + num1;
num1 = num2;
num2 = num3;
counter = counter + 1;
}

nFactorial Using Recursion:
public static void main(String args[]) {
int n = 5;
System.out.print(factRecur(n) + " ");
}

static int factRecur(int n) {
    int fact = 0;
    if(n == 0)
        return 1;
    else if (n > 0)
        fact = n * factRecur(n-1);
    else
        return -1;
    return fact;
}
